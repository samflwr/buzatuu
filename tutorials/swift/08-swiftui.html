swiftui state

import SwiftUI
struct ContentView: View {
  @State private var count = 0
  var body: some View {
    VStack {
      Text("count: \(count)")
      Button("tap") { count += 1 }
    }
  }
}

switch level {
case 1: print("base")
case 2: print("intermedio")
default: print("avanzato")
}
